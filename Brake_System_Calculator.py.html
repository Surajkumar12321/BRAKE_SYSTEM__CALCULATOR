def force_on_tmc(a11,a12):
  a1=a11*a12
  return(a1)
def area(c11):
  a2=0.785*c11*c11
  return(a2)
def braking_force(p1,a2,e11,f11):
  a3=(2*p1*a2*0.7*e11)/f11
  return(a3)
def braking_force1(p12,a22,e111,f111):
  t1=(p12*a22*0.7*e111)/f111
  return(t1)
def deceleration(t11,g11):
  a4=t11/g11
  return(a4)
def stoping_distance(h11,a41):
  a5=pow(h11,2)/(2*a41)
  return(a5)
def stoping_time(h11,a41):
  z5=h11/a41
  return(z5)
while True:
  print('''Select one option

            T_B_force =braking_force1(pressure1,area(i),j,f)+braking_force1(pressure1,area(i),j,f)

      print("enter_numenator_of_braking_weight_distribution_front_to_rear(like 63 for 63/37)")
